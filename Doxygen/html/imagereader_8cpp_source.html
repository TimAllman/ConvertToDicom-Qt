31 ImageReader::ImageVector ImageReader::ReadImage(const std::string& fileName)
32 {

37 
38  // If there s no suitable ITK IO type then quit
39  if (imageIO.IsNull())
40  {
41  LOG4CPLUS_ERROR(logger, "ImageIO instance not created from file: " << fileName);
42  return ImageVector();

50  typedef itk::ImageIOBase::IOComponentType ScalarPixelType;

55  LOG4CPLUS_DEBUG(logger, "numDimensions: " << numDimensions);

57  {
58  LOG4CPLUS_DEBUG(logger, " dimension[" << idx << "]: " << imageIO->GetDimensions(idx));
59  LOG4CPLUS_DEBUG(logger, " spacing[" << idx << "]: " << imageIO->GetSpacing(idx));

108 
109  for (unsigned sliceIdx = 0; sliceIdx < numSlices; ++sliceIdx)
110  {
111  // Generate the region that we want

115  sliceStart[2] = sliceIdx;

119

122  ExtractFiltertype::Pointer filter = ExtractFiltertype::New();
