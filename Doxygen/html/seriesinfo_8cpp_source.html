194 QString SeriesInfo::imagePositionPatientString(int sliceIdx) const
195 {
196  //LOG4CPLUS_TRACE(logger_, "Enter");
197 
198  vnl_matrix_fixed<double, 3, 3> rot; // the rotation matrix
199  vnl_vector_fixed<double, 3> ipp; // the IPP (column) vector
200 
201  // Create the rotation matrix from IOP
202  sscanf(m_imageOrientationPatient.toStdString().c_str(), "%lf\\%lf\\%lf\\%lf\\%lf\\%lf",
203  &rot(0, 0), &rot(0, 1), &rot(0, 2), &rot(1, 0), &rot(1, 1), &rot(1, 2));
204 
205  // Compute the remaining orthogonal vector to complete the matrix
206  rot(2,0) = rot(0, 1) * rot(1, 2) - rot(0, 2) * rot(1, 1);
207  rot(2,1) = rot(0, 2) * rot(1, 0) - rot(0, 0) * rot(1, 2);
208  rot(2,2) = rot(0, 0) * rot(1, 1) - rot(0, 1) * rot(1, 0);
209 
210  // IPP as a vector
211  ipp[0] = imagePositionPatientX();
212  ipp[1] = imagePositionPatientY();
213  ipp[2] = imagePositionPatientZ();
214 
215  LOG4CPLUS_DEBUG(m_logger, "Initial IPP = " << imagePositionPatientString().toStdString());
216 
217  ipp = rot * ipp; // rotate ipp into image coordinates
218  ipp[2] += m_imageSliceSpacing * sliceIdx; // increment Z component
219  ipp = rot.inplace_transpose() * ipp; // rotate back into patient coordinates
220 
221  char ippStr[30];
222  sprintf(ippStr, "%.2f\\%.2f\\%.2f", ipp(0), ipp(1), ipp(2));
223  LOG4CPLUS_DEBUG(m_logger, "Incremented IPP = " << ippStr);
224 
225  return ippStr;
226 
227  /* extracted from osirix, reflected in above code
228  float vec[3];
229  vec[0] = iop[1]*iop[5] - iop[2]*iop[4];
230  vec[1] = iop[2]*iop[3] - iop[0]*iop[5];
231  vec[2] = iop[0]*iop[4] - iop[1]*iop[3];
232  */
