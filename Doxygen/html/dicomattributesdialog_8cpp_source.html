127 void DicomAttributesDialog::loadData()
128 {
129  // Patient tab
130  ui->patientNameLineEdit->setText(seriesInfo->patientName());
131  ui->patientIDLineEdit->setText(seriesInfo->patientID());
132  ui->patientDOBDateEdit->setDate(seriesInfo->patientDOB());
133 
134  int idx = sexes.indexOf(seriesInfo->patientSex());
135  ui->patientSexComboBox->setCurrentIndex(idx);
136  LOG4CPLUS_DEBUG(logger, "Patient's sex set to " << ui->patientSexComboBox->currentText().toStdString());
137 
138  //Study tab
139  idx = modalities.indexOf(seriesInfo->studyModality());
140  ui->studyModalityComboBox->setCurrentIndex(idx);
141  LOG4CPLUS_DEBUG(logger, "Modality set to " << ui->studyModalityComboBox->currentText().toStdString());
142 
143  ui->studyDescriptionLineEdit->setText(seriesInfo->studyDescription());
144  ui->studyIDLineEdit->setText(seriesInfo->studyID());
145  ui->studyDateTimeDateTimeEdit->setDateTime(seriesInfo->studyDateTime());
146  ui->studyInstanceUIDLineEdit->setText(seriesInfo->studyInstanceUID());
147 
148  // Series tab
149  idx = positions.indexOf(seriesInfo->seriesPositionPatient());
150  ui->seriesPatientPositionComboBox->setCurrentIndex(idx);
151  LOG4CPLUS_DEBUG(logger, "Position set to "
152  << ui->seriesPatientPositionComboBox->currentText().toStdString());
153 
154  ui->seriesDescriptionLineEdit->setText(seriesInfo->seriesDescription());
155  ui->seriesNumberLineEdit->setText(seriesInfo->seriesNumberStr());
156  ui->seriesTimeIncrementLineEdit->setText(seriesInfo->seriesTimeIncrementStr());
157 
158  // Image tab
159  ui->numberOfImagesLineEdit->setText(QString::number(seriesInfo->imageNumberOfImages()));
160 
161  /* We need to set up the slices per image combobox.
162  * It is possible that a series of slices may be either single 2D images
163  * or slices of fewer 3D images. This code finds the list of exact divisors
164  * for the number of slices so that the DICOM images cam be written
165  * out properly.
166  */
167  int sliceCount = seriesInfo->imageNumberOfSlices();
168  QStringList countList;
169  for (int idx = 1; idx <= sliceCount; ++idx)
170  if ((sliceCount % idx) == 0)
171  countList.append(QString::number(idx));
172  int index = countList.indexOf(QString::number(seriesInfo->imageSlicesPerImage()));
173  if (index == -1)
174  index = 0;
175  //ui->imageSlicesPerImageComboBox->blockSignals(true);
176  ui->imageSlicesPerImageComboBox->clear();
177  ui->imageSlicesPerImageComboBox->addItems(countList);
178  //ui->imageSlicesPerImageComboBox->blockSignals(false);
179  ui->imageSlicesPerImageComboBox->setCurrentIndex(index);
180 
181  ui->imageSliceSpacingLineEdit->setText(QString::number(seriesInfo->imageSliceSpacing()));
182  ui->imagePatientPositionXLineEdit->setText(QString::number(seriesInfo->imagePositionPatientX()));
183  ui->imagePatientPositionYLineEdit->setText(QString::number(seriesInfo->imagePositionPatientY()));
184  ui->imagePatientPositionZLineEdit->setText(QString::number(seriesInfo->imagePositionPatientZ()));
185  ui->imagePatientOrientationLineEdit->setText(seriesInfo->imagePatientOrientation());
186 
187 }
