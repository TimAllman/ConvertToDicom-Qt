41 ErrorCode DicomSeriesWriter::WriteFileSeries()
42 {
43  LOG4CPLUS_TRACE(logger, "Enter");
44  // Set up the new metadata dictionary array
45  // This is based upon the example on the ITK examples wiki
46  // http://www.itk.org/Wiki/ITK/Examples/DICOM/ResampleDICOM
47  //
48 
49  PrepareMetaDataDictionaryArray();
50 
51  //
52  // create a Dicom series writer
53  itk::GDCMImageIO::Pointer dicomIo = itk::GDCMImageIO::New();
54  dicomIo->SetPixelType(itk::ImageIOBase::SCALAR);
55  dicomIo->KeepOriginalUIDOn();
56 
57  // define the filenames generator type and instance
58  typedef itk::NumericSeriesFileNames NameGeneratorType;
59  NameGeneratorType::Pointer nameGenerator = NameGeneratorType::New();
60 
61  QString value = outputDirectory + "/IM-" + QString::number(seriesInfo->seriesNumber()) + "-%04d.dcm";
62  nameGenerator->SetSeriesFormat(value.toStdString());
63  nameGenerator->SetStartIndex(1);
64  nameGenerator->SetEndIndex(itk::SizeValueType(images.size()));
65  fileNames = nameGenerator->GetFileNames();
66 
67  // We want to empty the output directory so we remove it and recreate it.
68  itksys::SystemTools::RemoveADirectory(outputDirectory.toStdString());
69  itksys::SystemTools::MakeDirectory(outputDirectory.toStdString());
70 
71  typedef itk::ImageSeriesWriter<Image3DType, Image2DType> WriterType;
72  WriterType::Pointer writer = WriterType::New();
73  writer->SetImageIO(dicomIo);
74  writer->SetFileNames(fileNames);
75  writer->SetMetaDataDictionaryArray(&dictArray);
76  Image3DType::Pointer image = MergeSlices();
77  writer->SetInput(image);
78 
79  try
80  {
81  writer->Update();
82  }
83  catch (itk::ExceptionObject& ex)
84  {
85  LOG4CPLUS_ERROR(logger, "ExceptionObject caught. " << ex.what());
86  return ErrorCode::ERROR_WRITING_FILE;
87  }
88 
89  return ErrorCode::SUCCESS;
90 }

92 void DicomSeriesWriter::CopyDictionary(const itk::MetaDataDictionary& fromDict,
93  itk::MetaDataDictionary& toDict)
94 {
95  LOG4CPLUS_TRACE(logger, "Enter");
96 
97  typedef itk::MetaDataDictionary DictionaryType;
98 
99  DictionaryType::ConstIterator itr = fromDict.Begin();
100  DictionaryType::ConstIterator end = fromDict.End();
101  typedef itk::MetaDataObject< std::string > MetaDataStringType;
102 
103  while (itr != end)
104  {
105  itk::MetaDataObjectBase::Pointer entry = itr->second;
106 
107  MetaDataStringType::Pointer entryvalue = dynamic_cast<MetaDataStringType *>(entry.GetPointer()) ;
108  if (entryvalue)
109  {
110  std::string tagkey = itr->first;
111  std::string tagvalue = entryvalue->GetMetaDataObjectValue();
112  itk::EncapsulateMetaData<std::string>(toDict, tagkey, tagvalue);
113  }
114  ++itr;
115  }
116 }

118 void DicomSeriesWriter::PrepareMetaDataDictionaryArray()
119 {
120  LOG4CPLUS_TRACE(logger, "Enter");
121 
122  // It may have been used in a previous run.
123  dictArray.clear();
124 
125  itk::MetaDataDictionary seriesDict = seriesInfo->metaDataDictionary();
126  LOG4CPLUS_TRACE(logger, "********** seriesDict - 1 ************");
127  LOG4CPLUS_TRACE(logger, DumpDicomMetaDataDictionary(seriesDict));
128 
129  bool isTimeSeries = (seriesInfo->seriesTimeIncrement() > 0.0);
130  if (isTimeSeries)
131  {
132  std::stringstream sstr;
133  int numTimes = seriesInfo->imageNumberOfImages();
134  if (numTimes > 1)
135  {
136  sstr.str("");
137  sstr << numTimes;
138  std::string numTemporalPositions = sstr.str();
139  itk::EncapsulateMetaData<std::string>(seriesDict, "0020|0105", numTemporalPositions);
140  }
141  }
142 
143  // These are converted images so we show that.
144  itk::EncapsulateMetaData<std::string>(seriesDict, "0008|0008", "ORIGINAL");
145  itk::EncapsulateMetaData<std::string>(seriesDict, "0008|0064", "WSD");
146 
147  LOG4CPLUS_TRACE(logger, "********** seriesDict - 2 ************");
148  LOG4CPLUS_TRACE(logger, DumpDicomMetaDataDictionary(seriesDict));
149 
150  // Derivation Description - How this image was derived
151  std::ostringstream value;
152  value.str("");
153  value << "Converted to DICOM using " << ITK_SOURCE_VERSION;
154  // Deal with a 1024 character max length.
155  unsigned lengthOfDesc = static_cast<unsigned>(value.str().length());
156  std::string derivationDesc(value.str(), 0, lengthOfDesc > 1024 ? 1024 : lengthOfDesc);
157  itk::EncapsulateMetaData<std::string>(seriesDict, "0008|2111", derivationDesc);
158 
159  // loop through the images, and the slices in each image
160  int instanceNumber = 1;
161  for (int imageIdx = 0; imageIdx < seriesInfo->imageNumberOfImages(); ++imageIdx)
162  {
163  itk::MetaDataDictionary imageDict;
164  CopyDictionary(seriesDict, imageDict);
165 
166  if (isTimeSeries)
167  {
168  // Temporal Position
169  std::stringstream sstr;
170  sstr.str("");
171  sstr << imageIdx+1;
172  std::string temporalPosition = sstr.str();
173  itk::EncapsulateMetaData<std::string>(imageDict, "0020|0100", temporalPosition);
174  }
175 
176  QTime time = seriesInfo->acqTimes()[imageIdx];
177  std::string acqTime = time.toString("HHmmss.zzz").toStdString();
178  itk::EncapsulateMetaData<std::string>(imageDict, "0008|0032", acqTime);
179 
180  float sliceLocation = 0.0;
181  for (int sliceIdx = 0; sliceIdx < seriesInfo->imageSlicesPerImage(); ++sliceIdx)
182  {
183  // Make a new dictionary for the slice and copy over the information already set
184  // We need a pointer because the dictionary array is an array of pointers.
185  itk::MetaDataDictionary *sliceDict = new itk::MetaDataDictionary();
186  CopyDictionary(imageDict, *sliceDict);
187 
188  gdcm::UIDGenerator sopuidGen;
189  std::string sopInstanceUID = sopuidGen.Generate();
190  //itk::EncapsulateMetaData<std::string>(*sliceDict, "0008|0018", sopInstanceUID);
191  itk::EncapsulateMetaData<std::string>(*sliceDict, "0002|0003", sopInstanceUID);
192 
193  // Set the IPP for this slice
194  std::stringstream sstr;
195  sstr << std::fixed << std::setprecision(2) << seriesInfo->imagePositionPatientX() << "\\"
196  << seriesInfo->imagePositionPatientY() << "\\" << seriesInfo->imagePositionPatientZ() << "\\";
197 
198  std::string imagePositionPatient = seriesInfo->imagePositionPatientString(sliceIdx).toStdString();
199  itk::EncapsulateMetaData<std::string>(*sliceDict, "0020|0032", imagePositionPatient);
200 
201  // The relative location of this slice from the first one.
202  sstr.str("");
203  sstr << std::fixed << std::setprecision(1) << sliceLocation;
204  itk::EncapsulateMetaData<std::string>(*sliceDict, "0020|1041", sstr.str());
205  sliceLocation += seriesInfo->imageSliceSpacing();
206 
207  sstr.str("");
208  sstr << instanceNumber;
209  itk::EncapsulateMetaData<std::string>(*sliceDict, "0020|0013", sstr.str());
210  ++instanceNumber;
211 
212  LOG4CPLUS_TRACE(logger, "*** Image " << imageIdx << " slice " << sliceIdx << " ***");
213  LOG4CPLUS_TRACE(logger, DumpDicomMetaDataDictionary(*sliceDict));
214  dictArray.push_back(sliceDict);
215  }
216  }
217 }

219 Image3DType::Pointer DicomSeriesWriter::MergeSlices()
220 {
221  LOG4CPLUS_TRACE(logger, "Enter");
222 
223  // Concatenate the whole bunch of slices into one image.
224  typedef itk::TileImageFilter<Image2DType, Image3DType> TileFilterType;
225  TileFilterType::Pointer tileFilter = TileFilterType::New();
226 
227  itk::FixedArray<unsigned int, 3> layout;
228  layout[0] = 1;
229  layout[1] = 1;
230  layout[2] = 0;
231 
232  LOG4CPLUS_DEBUG(logger, "Layout = " << layout);
233 
234  tileFilter->SetLayout(layout);
235  tileFilter->SetDefaultPixelValue(0);
236  int numSlices = images.size();
237  for (int idx = 0; idx < numSlices; ++idx)
238  {
239  tileFilter->SetInput(unsigned(idx), images[idx]);
240  }
241 
242  tileFilter->Update();
243 
244  Image3DType::Pointer image3d = tileFilter->GetOutput();
245  return image3d;
246 }
