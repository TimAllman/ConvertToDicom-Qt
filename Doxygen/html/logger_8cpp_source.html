33 void SetupLogger(const std::string& loggerName, LogLevel consoleLogLevel, LogLevel fileLogLevel,
34  const std::string& filePath)
35 {
36  static bool setup = false;
37  if (setup)
38  return;
39  setup = true;
40 
41  // The logger is set to log all messages. We use the appenders to restrict the output(s).
42  Logger logger = Logger::getInstance(loggerName);
43  logger.setLogLevel(int(LogLevel::LOG_LEVEL_ALL));
44 
45  // The console appender.
46  log4cplus::SharedAppenderPtr consoleAppender(new log4cplus::ConsoleAppender);
47  std::string consoleAppName = loggerName + ".console";
48  consoleAppender->setName(consoleAppName);
49  consoleAppender->setThreshold(int(consoleLogLevel));
50  //consoleAppender->setThreshold(int(LogLevel::LOG_LEVEL_INFO));
51  std::string consolePattern = "%-5p (%d{%q}) [%b:%L] %m%n";
52  std::auto_ptr<log4cplus::Layout> layout(new log4cplus::PatternLayout(consolePattern));
53  consoleAppender->setLayout(layout);
54 
55  logger.addAppender(consoleAppender);
56 
57  // Generate the name of the rolling file, by default in $HOME/Library/Logs
58  std::string logFilePath;
59  std::string logFileName = loggerName + ".log";
60  if (filePath.size() == 0)
61  {
62  #if defined(Q_OS_MACOS) || defined(Q_OS_IOS)
63  logFilePath = std::string(getenv("HOME")) + "/Library/Logs/";
64  #elif defined(Q_OS_WIN)
65  logFilePath = std::string(getenv("HOMEDRIVE")) + std::string(getenv("HOMEPATH")) + "\\Logs\\";
66  #elif defined(Q_OS_UNIX)
67  logFilePath = std::string("/var/log");
68  #endif
69  }
70  else
71  {
72  logFilePath = filePath + "/";
73  }
74 
75  logFilePath += logFileName;
76 
77  // Set up the Rolling File Appender format
78  std::string logFileAppName = loggerName + ".file";
79  log4cplus::SharedAppenderPtr
80  logFileApp(new log4cplus::RollingFileAppender(logFilePath, 1000000, 5, false));
81  logFileApp->setName(logFileAppName);
82  std::string filePattern = "%-5p [%d{%y-%m-%d %H:%M:%S:%q}][%b:%L] %m%n";
83  std::auto_ptr<log4cplus::Layout> fileLayout(new log4cplus::PatternLayout(filePattern));
84  logFileApp->setLayout(fileLayout);
85 
86  // Set up the log file level
87  int intFileLogLevel = int(fileLogLevel);
88  if (fileLogLevel == LogLevel::LOG_LEVEL_NOT_SET)
89  {
90  // Load logging level. If LoggingLevelKey is not found in the settings, the default
91  // value LogLevel::LOG_LEVEL_INFO is used.
92  Settings settings;
93  intFileLogLevel = settings.value(Settings::LoggingLevelKey, int(LogLevel::LOG_LEVEL_INFO)).toInt();
94  }
95 
96  logFileApp->setThreshold(intFileLogLevel);
97  logger.addAppender(logFileApp);
98 
99  // Force this to the console.
100  LOG4CPLUS_INFO(logger, "Logging to file: " << logFilePath << ", Level: "
101  << LogLevelToString(LogLevel(intFileLogLevel)));
102 }

115 std::string LogLevelToString(LogLevel level)
116 {
117  std::string retVal;
118 
119  switch (level)
120  {
121  case LogLevel::LOG_LEVEL_OFF:
122  retVal = "OFF";
123  break;
124  case LogLevel::LOG_LEVEL_FATAL:
125  retVal = "FATAL";
126  break;
127  case LogLevel::LOG_LEVEL_ERROR:
128  retVal = "ERROR";
129  break;
130  case LogLevel::LOG_LEVEL_WARN:
131  retVal = "WARN";
132  break;
133  case LogLevel::LOG_LEVEL_INFO:
134  retVal = "INFO";
135  break;
136  case LogLevel::LOG_LEVEL_DEBUG:
137  retVal = "DEBUG";
138  break;
139  case LogLevel::LOG_LEVEL_TRACE:
140  retVal = "TRACE";
141  break;
142  case LogLevel::LOG_LEVEL_NOT_SET:
143  retVal = "NOT SET";
144  break;
145  }
146 
147  return retVal;
148 }
